<?php
// Giả sử đã có kết nối cơ sở dữ liệu $pdo

// Bắt đầu transaction
$pdo->beginTransaction();

try {
    // 1. Lấy thông tin giỏ hàng
    $customerID = $_SESSION['CustomerID']; // Giả sử ID khách hàng đã lưu trong session
    $cartQuery = "SELECT BookID, Quantity FROM Cart WHERE CustomerID = :customerID";
    $stmt = $pdo->prepare($cartQuery);
    $stmt->execute(['customerID' => $customerID]);
    $cartItems = $stmt->fetchAll();

    if (empty($cartItems)) {
        throw new Exception("Giỏ hàng trống!");
    }

    // 2. Tính tổng số tiền đơn hàng
    $totalAmount = 0;
    foreach ($cartItems as $item) {
        $bookID = $item['BookID'];
        $quantity = $item['Quantity'];

        // Lấy giá của sách từ bảng Books
        $bookQuery = "SELECT Price, StockQuantity FROM Books WHERE BookID = :bookID";
        $stmt = $pdo->prepare($bookQuery);
        $stmt->execute(['bookID' => $bookID]);
        $book = $stmt->fetch();

        // Kiểm tra số lượng trong kho
        if ($book['StockQuantity'] < $quantity) {
            throw new Exception("Sản phẩm không đủ trong kho!");
        }

        // Cộng dồn tổng tiền
        $totalAmount += $book['Price'] * $quantity;
    }

    // 3. Tạo đơn hàng mới
    $orderQuery = "INSERT INTO Orders (CustomerID, TotalAmount, Status) VALUES (:customerID, :totalAmount, 'Completed')";
    $stmt = $pdo->prepare($orderQuery);
    $stmt->execute(['customerID' => $customerID, 'totalAmount' => $totalAmount]);
    $orderID = $pdo->lastInsertId(); // Lấy ID của đơn hàng vừa tạo

    // 4. Lưu chi tiết đơn hàng vào OrderDetails
    foreach ($cartItems as $item) {
        $bookID = $item['BookID'];
        $quantity = $item['Quantity'];

        // Lưu chi tiết đơn hàng
        $orderDetailQuery = "INSERT INTO OrderDetails (OrderID, BookID, Quantity, Price) VALUES (:orderID, :bookID, :quantity, :price)";
        $stmt = $pdo->prepare($orderDetailQuery);
        $stmt->execute([
            'orderID' => $orderID,
            'bookID' => $bookID,
            'quantity' => $quantity,
            'price' => $book['Price']
        ]);

        // 5. Cập nhật số lượng trong kho
        $updateStockQuery = "UPDATE Books SET StockQuantity = StockQuantity - :quantity WHERE BookID = :bookID";
        $stmt = $pdo->prepare($updateStockQuery);
        $stmt->execute([
            'quantity' => $quantity,
            'bookID' => $bookID
        ]);
    }

    // 6. Xóa giỏ hàng
    $deleteCartQuery = "DELETE FROM Cart WHERE CustomerID = :customerID";
    $stmt = $pdo->prepare($deleteCartQuery);
    $stmt->execute(['customerID' => $customerID]);

    // Commit transaction
    $pdo->commit();

    echo "Thanh toán thành công!";
} catch (Exception $e) {
    // Rollback transaction nếu có lỗi
    $pdo->rollBack();
    echo "Lỗi: " . $e->getMessage();
}
?>

<?php

    header("Location: index.php?controller=auth&action=login"); // Chuyển hướng về trang đăng nhập
    exit();

//Database connection
$host = "localhost";
$username = "root";
$password = "";
$dbname = "bookstore";
$conn = new mysqli($host, $username, $password, $dbname);
if ($conn->connect_error) {
    die("Connection failed: Please try again later.");
}

// Fetch authors and categories for the dropdowns
$authors = [];
$categories = [];

$authorResult = $conn->query("SELECT AuthorID, Name FROM Authors");
if ($authorResult->num_rows > 0) {
    while ($row = $authorResult->fetch_assoc()) {
        $authors[] = $row;
    }
}

$categoryResult = $conn->query("SELECT CategoryID, CategoryName FROM Categories");
if ($categoryResult->num_rows > 0) {
    while ($row = $categoryResult->fetch_assoc()) {
        $categories[] = $row;
    }
}
// Check if the form is submitted
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Retrieve form data
    $title = $_POST['title'];
    $author_id = $_POST['author_id'];
    $category_id = $_POST['category_id'];
    $price = $_POST['price'];
    $quantity = $_POST['quantity'];
    $published_date = $_POST['published_date'];
    $description = $_POST['description'];
$cover_image_url = null;
if (!empty($_FILES['cover_image']['name'])) {
    // Đường dẫn đến thư mục chứa ảnh
    $target_dir = $_SERVER['DOCUMENT_ROOT'] . "/teststore/assets/uploads/";
    
    // Lấy tên file ảnh
    $cover_image_name = basename($_FILES["cover_image"]["name"]);
    
    // Đường dẫn đầy đủ của ảnh để lưu trữ
    $cover_image_url = "assets/uploads/" . $cover_image_name;
    
    // Di chuyển ảnh từ thư mục tạm vào thư mục uploads
    if (!move_uploaded_file($_FILES["cover_image"]["tmp_name"], $target_dir . $cover_image_name)) {
        echo "<p>Failed to upload cover image to $cover_image_url. Please check your server permissions.</p>";
        exit;
    }
}

// Insert dữ liệu vào cơ sở dữ liệu
$stmt = $conn->prepare("INSERT INTO Books (Title, AuthorID, CategoryID, Price, Quantity, PublishedDate, Description, CoverImageUrl) VALUES (?, ?, ?, ?, ?, ?, ?, ?)");
$stmt->bind_param("siiidsss", $title, $author_id, $category_id, $price, $quantity, $published_date, $description, $cover_image_url);

// Execute and check for success
if ($stmt->execute()) {
    echo "<p>Thêm sách mới thành công!</p>";
} else {
    echo "<p>Đã xảy ra lỗi khi thêm sách: " . $stmt->error . "</p>";
}
}
$conn->close();
